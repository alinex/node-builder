  var async, chalk, coffeelint, coverage, coveralls, debug, exec, execFile, fs, openUrl, path, ref, spawn, testMocha;

  debug = require(&#039;debug&#039;)(&#039;builder:test&#039;);

  async = require(&#039;async&#039;);

  fs = require(&#039;alinex-fs&#039;);

  path = require(&#039;path&#039;);

  chalk = require(&#039;chalk&#039;);

  ref = require(&#039;child_process&#039;), spawn = ref.spawn, exec = ref.exec, execFile = ref.execFile;

  module.exports.run = function(dir, options, cb) {
    return coffeelint(dir, options, function(err) {
      if (err) {
        return cb(err);
      }
      return testMocha(dir, options, function(err) {
        if (err) {
          return cb(err);
        }
        return coverage(dir, options, function(err) {
          var url;
          if (err) {
            return cb(err);
          }
          url = path.join(GLOBAL.ROOT_DIR, dir, &#039;coverage&#039;, &#039;lcov-report&#039;, &#039;index.html&#039;);
          if (options.coverage &amp;&amp; options.browser) {
            return openUrl(options, url, cb);
          }
          return cb();
        });
      });
    });
  };

  openUrl = function(options, target, cb) {
    var opener;
    if (options.verbose) {
      console.log(chalk.grey(&quot;Open &quot; + target + &quot; in browser&quot;));
    }
    opener = (function() {
      switch (process.platform) {
        case &#039;darwin&#039;:
          return &#039;open&#039;;
        case &#039;win32&#039;:
          return &#039;start &quot;&quot;&#039;;
        default:
          return path.join(GLOBAL.ROOT_DIR, &#039;bin/xdg-open&#039;);
      }
    })();
    debug(&quot;exec&gt; &quot; + opener + &quot; \&quot;&quot; + (encodeURI(target)) + &quot;\&quot;&quot;);
    return exec(opener + &#039; &quot;&#039; + encodeURI(target) + &#039;&quot;&#039;, cb);
  };

  coffeelint = function(dir, options, cb) {
    return fs.npmbin(&#039;coffeelint&#039;, path.dirname(path.dirname(__dirname)), function(err, cmd) {
      var proc;
      if (err) {
        console.log(chalk.yellow(&quot;Skipped lint because coffeelint is missing&quot;));
        return typeof cb === &quot;function&quot; ? cb() : void 0;
      }
      console.log(&quot;Linting coffee code&quot;);
      debug(&quot;exec &quot; + dir + &quot;&gt; &quot; + cmd + &quot; -f &quot; + (path.join(GLOBAL.ROOT_DIR, &#039;coffeelint.json&#039;)) + &quot; src&quot;);
      if (options.nocolors) {
        proc = spawn(cmd, [&#039;-f&#039;, path.join(GLOBAL.ROOT_DIR, &#039;coffeelint.json&#039;), &#039;src&#039;], {
          cwd: dir,
          stdio: &#039;inherit&#039;
        });
      } else {
        proc = spawn(cmd, [&#039;-f&#039;, path.join(GLOBAL.ROOT_DIR, &#039;coffeelint.json&#039;), &#039;src&#039;], {
          cwd: dir
        });
        proc.stdout.on(&#039;data&#039;, function(data) {
          if (options.verbose) {
            return console.log(data.toString().trim());
          }
        });
        proc.stderr.on(&#039;data&#039;, function(data) {
          return console.error(chalk.magenta(data.toString().trim()));
        });
      }
      proc.on(&#039;error&#039;, cb);
      return proc.on(&#039;exit&#039;, function(status) {
        if (status !== 0) {
          status = &quot;Coffeelint exited with status &quot; + status;
        }
        return cb(status);
      });
    });
  };

  testMocha = function(dir, options, cb) {
    var mochadir;
    mochadir = path.join(dir, &#039;test&#039;, &#039;mocha&#039;);
    if (!fs.existsSync(mochadir)) {
      console.log(chalk.magenta(&quot;No mocha test dir found at &quot; + mochadir + &quot;.&quot;));
      return cb();
    }
    return fs.npmbin(&#039;istanbul&#039;, path.dirname(path.dirname(__dirname)), function(err, cmd) {
      var mocha;
      if (err) {
        return cb(err);
      }
      mocha = options.coverage ? &#039;_mocha&#039; : &#039;mocha&#039;;
      return fs.npmbin(mocha, path.dirname(path.dirname(__dirname)), function(err, mocha) {
        var args, proc;
        if (err) {
          return cb(err);
        }
        console.log(&quot;Run mocha tests&quot;);
        args = [];
        if (options.coverage) {
          args.push(&#039;cover&#039;, mocha, &#039;--&#039;, &#039;--require&#039;, &#039;coffee-coverage/register-istanbul&#039;);
        } else {
          cmd = mocha;
          if (options.watch) {
            args.push(&#039;-w&#039;);
          }
        }
        args.push(&#039;--compilers&#039;, &#039;coffee:coffee-script/register&#039;);
        args.push(&#039;--reporter&#039;, &#039;spec&#039;);
        args.push(&#039;-c&#039;);
        args.push(&#039;--recursive&#039;);
        args.push(&#039;test/mocha&#039;);
        debug(&quot;exec &quot; + dir + &quot;&gt; &quot; + cmd + &quot; &quot; + (args.join(&#039; &#039;)));
        proc = spawn(cmd, args, {
          cwd: dir,
          stdio: &#039;inherit&#039;,
          env: process.env
        });
        proc.on(&#039;error&#039;, cb);
        return proc.on(&#039;exit&#039;, function(status) {
          if (status !== 0) {
            status = &quot;Test exited with status &quot; + status;
          }
          return cb(status);
        });
      });
    });
  };

  coverage = function(dir, options, cb) {
    var mochadir;
    if (!options.coverage) {
      return cb();
    }
    mochadir = path.join(dir, &#039;test&#039;, &#039;mocha&#039;);
    if (!fs.existsSync(mochadir)) {
      return cb(&quot;Coverage only works on mocha tests.&quot;);
    }
    return fs.npmbin(&#039;istanbul&#039;, path.dirname(path.dirname(__dirname)), function(err, cmd) {
      var proc;
      if (err) {
        return cb(err);
      }
      console.log(&quot;Create coverage report&quot;);
      debug(&quot;exec &quot; + dir + &quot;&gt; &quot; + cmd + &quot; report&quot;);
      proc = spawn(cmd, [&#039;report&#039;], {
        cwd: dir
      });
      if (options.verbose) {
        proc.stderr.on(&#039;data&#039;, function(data) {
          return console.error(chalk.grey(data.toString().trim()));
        });
      }
      proc.on(&#039;error&#039;, cb);
      return proc.on(&#039;exit&#039;, function(status) {
        if (status !== 0) {
          return cb(&quot;Istanbul exited with status &quot; + status);
        }
        if (options.coveralls) {
          return coveralls(dir, options, cb);
        }
        return cb();
      });
    });
  };

  coveralls = function(dir, options, cb) {
    var file;
    file = path.join(dir, &#039;coverage&#039;, &#039;lcov.info&#039;);
    coveralls = path.join(GLOBAL.ROOT_DIR, &#039;node_modules/coveralls/bin/coveralls.js&#039;);
    debug(&quot;exec&gt; cat &quot; + file + &quot; | &quot; + coveralls + &quot; --verbose&quot;);
    return exec(&quot;cat &quot; + file + &quot; | &quot; + coveralls + &quot; --verbose&quot;, function(err, stdout, stderr) {
      if (stdout) {
        console.log(stdout.toString().trim());
      }
      return cb(err);
    });
  };

}).call(this);
