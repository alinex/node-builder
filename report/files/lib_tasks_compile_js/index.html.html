    <textarea id="file-source" class="col-md-12">(function() {
  var Spawn, async, chalk, coffee, compileCoffee, compileMan, fs, path, uglify;

  fs = require(&#039;alinex-fs&#039;);

  Spawn = require(&#039;alinex-spawn&#039;);

  path = require(&#039;path&#039;);

  async = require(&#039;async&#039;);

  chalk = require(&#039;chalk&#039;);

  coffee = require(&#039;coffee-script&#039;);

  module.exports.run = function(dir, options, cb) {
    var src;
    src = path.join(dir, &#039;src&#039;);
    if (!fs.existsSync(src)) {
      return cb(&quot;No source files found.&quot;);
    }
    console.log(&quot;Remove old directories&quot;);
    return async.each([path.join(dir, &#039;lib&#039;), path.join(dir, &#039;man&#039;)], fs.remove, function(err) {
      if (err) {
        return cb(err);
      }
      return async.parallel([
        function(cb) {
          return compileCoffee(dir, options, cb);
        }, function(cb) {
          return compileMan(dir, options, cb);
        }
      ], cb);
    });
  };

  compileCoffee = function(dir, options, cb) {
    var lib, src;
    src = path.join(dir, &#039;src&#039;);
    lib = path.join(dir, &#039;lib&#039;);
    return fs.find(src, {
      include: &#039;*.coffee&#039;
    }, function(err, files) {
      if (err) {
        return cb(err);
      }
      if (!files) {
        return cb();
      }
      console.log(&quot;Compile coffee script&quot;);
      return async.each(files, function(file, cb) {
        return fs.readFile(file, &#039;utf8&#039;, function(err, data) {
          var compiled, filepathjs, filepathmap, jsfile, mapfile;
          if (err) {
            return cb(err);
          }
          jsfile = path.basename(file, &#039;.coffee&#039;) + &#039;.js&#039;;
          mapfile = path.basename(file, &#039;.coffee&#039;) + &#039;.map&#039;;
          if (options.verbose) {
            console.log(chalk.grey(&quot;Compile &quot; + file));
          }
          compiled = coffee.compile(data, {
            filename: path.basename(file),
            generatedFile: jsfile,
            sourceRoot: path.relative(path.dirname(file), dir),
            sourceFiles: [file],
            sourceMap: true
          });
          compiled.js += &quot;\n//# sourceMappingURL=&quot; + (path.basename(mapfile));
          filepathjs = path.join(lib, path.dirname(file).slice(src.length), jsfile);
          filepathmap = path.join(lib, path.dirname(file).slice(src.length), mapfile);
          return async.parallel([
            function(cb) {
              return fs.mkdirs(path.dirname(filepathjs), function(err) {
                if (err &amp;&amp; err.code !== &#039;EEXIST&#039;) {
                  return cb(err);
                }
                return fs.writeFile(filepathjs, compiled.js, cb);
              });
            }, function(cb) {
              return fs.mkdirs(path.dirname(filepathmap), function(err) {
                if (err &amp;&amp; err.code !== &#039;EEXIST&#039;) {
                  return cb(err);
                }
                return fs.writeFile(filepathmap, compiled.v3SourceMap, cb);
              });
            }
          ], function(err) {
            if (err || !options.uglify) {
              return cb(err);
            }
            if (options.verbose) {
              console.log(chalk.grey(&quot;Uglify &quot; + jsfile));
            }
            return uglify({
              dir: path.dirname(filepathjs),
              fromjs: jsfile,
              frommap: mapfile,
              tojs: jsfile,
              tomap: mapfile
            }, cb);
          });
        });
      }, cb);
    });
  };

  compileMan = function(dir, options, cb) {
    var err, file, pack, src;
    file = path.join(dir, &#039;package.json&#039;);
    try {
      pack = JSON.parse(fs.readFileSync(file));
    } catch (_error) {
      err = _error;
      return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
    }
    if (pack.man == null) {
      return cb();
    }
    console.log(&quot;Compile man pages&quot;);
    src = path.join(dir, &#039;src&#039;);
    return fs.mkdirs(path.join(dir, &#039;man&#039;), function(err) {
      if (err) {
        return cb(err);
      }
      if (typeof pack.man === &#039;string&#039;) {
        pack.man = [pack.man];
      }
      return async.each(pack.man, function(name, cb) {
        var input;
        input = src + &quot;/&quot; + name + &quot;.md&quot;;
        return fs.exists(input, function(exists) {
          if (!exists) {
            return cb(new Error(&quot;The file &#039;&quot; + input + &quot;&#039; didn&#039;t exist.&quot;));
          }
          if (options.verbose) {
            console.log(chalk.grey(&quot;Create &quot; + pack.man));
          }
          return fs.npmbin(&#039;marked-man&#039;, path.dirname(path.dirname(__dirname)), function(err, cmd) {
            var proc;
            if (err) {
              return cb(err);
            }
            proc = new Spawn({
              priority: 9,
              cmd: cmd,
              args: [input]
            });
            return proc.run(function(err, stdout, stderr, code) {
              if (err) {
                return cb(err);
              }
              return fs.writeFile(path.join(dir, name), stdout, cb);
            });
          });
        });
      }, cb);
    });
  };

  uglify = function(item, cb) {
    return fs.npmbin(&#039;uglifyjs&#039;, path.dirname(path.dirname(__dirname)), function(err, cmd) {
      var args, proc;
      if (err) {
        return cb(err);
      }
      args = [item.fromjs, &#039;--source-map&#039;, item.tomap, &#039;-o&#039;, item.tojs];
      if (item.frommap) {
        args.push(&#039;--in-source-map&#039;, item.frommap);
      }
      proc = new Spawn({
        priority: 9,
        cmd: cmd,
        args: args,
        cwd: item.dir
      });
      return proc.run(function(err, stdout, stderr, code) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        return cb(err);
      });
    });
  };

}).call(this);
