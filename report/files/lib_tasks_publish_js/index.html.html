    <textarea id="file-source" class="col-md-12">(function() {
  var async, chalk, checkTests, commitChanges, debug, execFile, fs, gitTag, moment, path, pushNpm, pushOrigin, request, updateChangelog;

  debug = require(&#039;debug&#039;)(&#039;make:publish&#039;);

  async = require(&#039;async&#039;);

  path = require(&#039;path&#039;);

  chalk = require(&#039;chalk&#039;);

  execFile = require(&#039;child_process&#039;).execFile;

  request = require(&#039;request&#039;);

  moment = require(&#039;moment&#039;);

  fs = require(&#039;alinex-fs&#039;);

  module.exports.run = function(dir, options, cb) {
    return checkTests(dir, options, function(err) {
      var file, pack, version;
      if (err) {
        return cb(err);
      }
      file = path.join(dir, &#039;package.json&#039;);
      try {
        pack = JSON.parse(fs.readFileSync(file));
      } catch (_error) {
        err = _error;
        return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
      }
      options.oldVersion = pack.version;
      if (options.verbose) {
        console.log(chalk.grey(&quot;Old version is &quot; + options.oldVersion));
      }
      version = pack.version.split(/\./);
      if (options.major) {
        version[0]++;
        version[1] = version[2] = 0;
      } else if (options.minor) {
        version[1]++;
        version[2] = 0;
      } else {
        version[2]++;
      }
      options.newVersion = pack.version = version.join(&#039;.&#039;);
      if (options.verbose) {
        console.log(chalk.grey(&quot;New version is &quot; + pack.version));
      }
      console.log(&quot;Change package.json&quot;);
      return fs.writeFile(file, JSON.stringify(pack, null, 2), function(err) {
        if (err) {
          return cb(err);
        }
        return async.series([
          function(cb) {
            return updateChangelog(dir, options, cb);
          }, function(cb) {
            return commitChanges(dir, options, cb);
          }, function(cb) {
            return pushOrigin(dir, options, cb);
          }, function(cb) {
            return gitTag(dir, options, cb);
          }, function(cb) {
            return pushNpm(dir, options, cb);
          }
        ], function(err) {
          if (err) {
            throw err;
          }
          console.log(chalk.green(&quot;Created v&quot; + pack.version + &quot;.&quot;));
          return cb();
        });
      });
    });
  };

  checkTests = function(dir, options, cb) {
    if (options.verbose) {
      console.log(chalk.grey(&quot;Search for .only mocha tests...&quot;));
    }
    return fs.find(path.join(dir, &#039;test&#039;, &#039;mocha&#039;), {
      type: &#039;file&#039;
    }, function(err, list) {
      if (err) {
        return cb(err);
      }
      return async.each(list, function(file, cb) {
        return fs.readFile(file, &#039;utf-8&#039;, function(err, content) {
          if (err) {
            return err;
          }
          if (content.match(/(describe|it)\.only/)) {
            return cb(file);
          }
          return cb();
        });
      }, function(err, file) {
        if (!err) {
          return cb();
        }
        if (typeof err === &#039;string&#039;) {
          return cb(new Error(&quot;Tests with .only were found in &quot; + err + &quot;. But a full test is neccessary to publish.&quot;));
        }
        return cb(err);
      });
    });
  };

  updateChangelog = function(dir, options, cb) {
    var args;
    if (options.verbose) {
      console.log(chalk.grey(&quot;Read git log&quot;));
    }
    args = [&#039;log&#039;, &#039;--pretty=format:%s&#039;];
    if (options.oldVersion !== &#039;0.0.0&#039;) {
      args.push(&quot;v&quot; + options.oldVersion + &quot;..HEAD&quot;);
    }
    debug(&quot;exec &quot; + dir + &quot;&gt; git &quot; + (args.join(&#039; &#039;)));
    return execFile(&quot;git&quot;, args, {
      cwd: dir
    }, function(err, stdout, stderr) {
      var changelog, file, lines, newlines;
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stderr.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (err) {
        return cb(err);
      }
      file = path.join(dir, &#039;Changelog.md&#039;);
      lines = fs.readFileSync(file, &#039;utf-8&#039;).split(/\n/);
      newlines = stdout.trim().split(/\n/).map(function(val) {
        return &quot;- &quot; + val;
      });
      changelog = lines.slice(0, 6).join(&#039;\n&#039;) + (&quot;\nVersion &quot; + options.newVersion + &quot; (&quot; + (moment().format(&#039;YYYY-MM-DD&#039;)) + &quot;)\n-------------------------------------------------\n&quot; + (newlines.join(&#039;\n&#039;)) + &quot;\n&quot;) + lines.slice(5).join(&#039;\n&#039;);
      console.log(&quot;Write new changelog&quot;);
      return fs.writeFile(file, changelog, function(err) {
        if (err) {
          return cb(err);
        }
        return cb();
      });
    });
  };

  commitChanges = function(dir, options, cb) {
    console.log(&quot;Commit new version information&quot;);
    debug(&quot;exec &quot; + dir + &quot;&gt; git add package.json Changelog.md&quot;);
    return execFile(&quot;git&quot;, [&#039;add&#039;, &#039;package.json&#039;, &#039;Changelog.md&#039;], {
      cwd: dir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (err) {
        return cb(err);
      }
      debug(&quot;exec &quot; + dir + &quot;&gt; git commit \&quot;Added information for version &quot; + options.newVersion + &quot;\&quot;&quot;);
      return execFile(&quot;git&quot;, [&#039;commit&#039;, &#039;-m&#039;, &quot;Added information for version &quot; + options.newVersion], {
        cwd: dir
      }, function(err, stdout, stderr) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        if (err) {
          return cb(err);
        }
        return cb();
      });
    });
  };

  pushOrigin = function(dir, options, cb) {
    console.log(&quot;Push to git origin&quot;);
    debug(&quot;exec &quot; + dir + &quot;&gt; git push origin master&quot;);
    return execFile(&quot;git&quot;, [&#039;push&#039;, &#039;origin&#039;, &#039;master&#039;], {
      cwd: dir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (err) {
        return cb(err);
      }
      return cb();
    });
  };

  gitTag = function(dir, options, cb) {
    var changelog, file, pack;
    file = path.join(dir, &#039;package.json&#039;);
    pack = JSON.parse(fs.readFileSync(file));
    changelog = &#039;&#039;;
    if (pack.homepage != null) {
      changelog = &quot; see more in [Changelog.md](&quot; + pack.homepage + &quot;/Changelog.md.html)&quot;;
    }
    console.log(&quot;Push new tag to git origin&quot;);
    debug(&quot;exec &quot; + dir + &quot;&gt; git tag -a v&quot; + options.newVersion + &quot; -m \&quot;Created version &quot; + options.newVersion + changelog + &quot;\&quot;&quot;);
    return execFile(&quot;git&quot;, [&#039;tag&#039;, &#039;-a&#039;, &quot;v&quot; + options.newVersion, &#039;-m&#039;, &quot;Created version &quot; + options.newVersion + changelog], {
      cwd: dir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (err) {
        return cb(err);
      }
      debug(&quot;exec &quot; + dir + &quot;&gt; git push origin v&quot; + options.newVersion);
      return execFile(&quot;git&quot;, [&#039;push&#039;, &#039;origin&#039;, &quot;v&quot; + options.newVersion], {
        cwd: dir
      }, function(err, stdout, stderr) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        if (err) {
          return cb(err);
        }
        return cb();
      });
    });
  };

  pushNpm = function(dir, options, cb) {
    console.log(&quot;Push to npm&quot;);
    debug(&quot;exec &quot; + dir + &quot;&gt; npm install&quot;);
    return execFile(&#039;npm&#039;, [&#039;install&#039;], {
      cwd: dir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (err) {
        return cb(err);
      }
      debug(&quot;exec &quot; + dir + &quot;&gt; npm publish&quot;);
      return execFile(&#039;npm&#039;, [&#039;publish&#039;], {
        cwd: dir
      }, function(err, stdout, stderr) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        if (err) {
          return cb(err);
        }
        return cb();
      });
    });
  };

}).call(this);
