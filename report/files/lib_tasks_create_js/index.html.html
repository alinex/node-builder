    <textarea id="file-source" class="col-md-12">(function() {
  var async, chalk, createChangelog, createDir, createGitHub, createPackage, createReadme, createTravis, debug, execFile, fs, initGit, initialCommit, path, prompt, request;

  debug = require(&#039;debug&#039;)(&#039;make:create&#039;);

  async = require(&#039;async&#039;);

  fs = require(&#039;alinex-fs&#039;);

  path = require(&#039;path&#039;);

  chalk = require(&#039;chalk&#039;);

  execFile = require(&#039;child_process&#039;).execFile;

  request = require(&#039;request&#039;);

  prompt = require(&#039;prompt&#039;);

  module.exports.run = function(dir, options, cb) {
    prompt.start();
    return async.series([
      function(cb) {
        return createDir(dir, options, cb);
      }, function(cb) {
        return initGit(dir, options, cb);
      }, function(cb) {
        return createGitHub(dir, options, cb);
      }, function(cb) {
        return createPackage(dir, options, cb);
      }, function(cb) {
        return createTravis(dir, options, cb);
      }, function(cb) {
        return createReadme(dir, options, cb);
      }, function(cb) {
        return createChangelog(dir, options, cb);
      }, function(cb) {
        return initialCommit(dir, options, cb);
      }
    ], function(err) {
      if (err) {
        throw err;
      }
      console.log(chalk.yellow(&quot;You may now work with the new package.&quot;));
      return cb();
    });
  };

  createDir = function(dir, options, cb) {
    if (fs.existsSync(dir)) {
      if (options.verbose) {
        console.log(chalk.grey(&quot;Directory &quot; + dir + &quot; already exists.&quot;));
      }
      return cb();
    }
    console.log(&quot;Create directory &quot; + dir);
    return fs.mkdirs(path.join(dir, &#039;src&#039;), function(err) {
      if (err) {
        return cb(err);
      }
      return fs.mkdirs(path.join(dir, &#039;test&#039;, &#039;mocha&#039;), function(err) {
        var file;
        if (err) {
          return cb(err);
        }
        file = path.join(dir, &#039;.npmignore&#039;);
        return fs.copy(path.join(GLOBAL.ROOT_DIR, &#039;.npmignore&#039;), file, cb);
      });
    });
  };

  initGit = function(dir, options, cb) {
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for configured git&quot;));
    }
    if (fs.existsSync(path.join(dir, &#039;.git&#039;))) {
      options.git = &#039;file://&#039; + fs.realpathSync(dir);
      return cb();
    }
    return prompt.get({
      message: &quot;Should a git repository be initialized?&quot;,
      validator: /y[es]*|n[o]?/,
      warning: &#039;You must respond with yes or no&#039;,
      &quot;default&quot;: &#039;yes&#039;
    }, function(err, input) {
      if (err || input.question !== &#039;yes&#039;) {
        return cb(err);
      }
      console.log(&quot;Init new git repository&quot;);
      debug(&quot;exec &quot; + dir + &quot;&gt; git init&quot;);
      return execFile(&quot;git&quot;, [&#039;init&#039;], {
        cwd: dir
      }, function(err, stdout, stderr) {
        var file;
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        file = path.join(dir, &#039;.gitignore&#039;);
        if (err || fs.existsSync(file)) {
          return cb(err);
        }
        options.git = &#039;file://&#039; + fs.realpathSync(dir);
        return fs.copy(path.join(GLOBAL.ROOT_DIR, &#039;.gitignore&#039;), file, cb);
      });
    });
  };

  createGitHub = function(dir, options, cb) {
    var err, file, pack;
    if (options[&quot;private&quot;]) {
      return cb();
    }
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for configured git&quot;));
    }
    file = path.join(dir, &#039;package.json&#039;);
    if (fs.existsSync(file)) {
      try {
        pack = JSON.parse(fs.readFileSync(file));
      } catch (_error) {
        err = _error;
        return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
      }
      if (pack.repository.type !== &#039;git&#039;) {
        console.out(chalk.yellow(&quot;Only git repositories can be added to github.&quot;));
        return cb();
      }
      console.log(pack.repository.url);
      console.log(~pack.repository.url.indexOf(&#039;github.com/&#039;));
      if (~pack.repository.url.indexOf(&#039;github.com/&#039;)) {
        options.github = pack.repository.url;
        return cb();
      }
    }
    debug(&quot;exec &quot; + dir + &quot;&gt; git remote show origin&quot;);
    return execFile(&quot;git&quot;, [&#039;remote&#039;, &#039;show&#039;, &#039;origin&#039;], {
      cwd: dir
    }, function(err, stdout, stderr) {
      if (!err) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        console.log(&quot;Skipped GitHub because other origin exists already&quot;);
        return cb();
      }
      return prompt.get({
        message: &quot;Should a github repository be initialized?&quot;,
        validator: /y[es]*|n[o]?/,
        warning: &#039;You must respond with yes or no&#039;,
        &quot;default&quot;: &#039;yes&#039;
      }, function(err, input) {
        console.log(input);
        if (err || input.question !== &#039;yes&#039;) {
          return cb(err);
        }
        console.log(&quot;Init new git repository&quot;);
        return prompt.get([
          {
            message: &quot;GitHub username:&quot;,
            name: &#039;username&#039;,
            required: true,
            &quot;default&quot;: &#039;alinex&#039;
          }, {
            message: &quot;Password for GitHub login:&quot;,
            name: &#039;password&#039;,
            hidden: true,
            required: true
          }, {
            message: &quot;Give a short description of this module.&quot;,
            name: &#039;description&#039;,
            required: true
          }
        ], function(err, input) {
          var gitname;
          gitname = path.basename(dir);
          options.github = &quot;https://github.com/&quot; + input.username + &quot;/&quot; + gitname;
          return request({
            uri: &quot;https://api.github.com/repos/&quot; + input.username + &quot;/&quot; + gitname,
            auth: {
              user: input.username,
              pass: input.password
            },
            headers: {
              &#039;User-Agent&#039;: input.username
            }
          }, function(err, response, body) {
            var answer;
            if (err) {
              return cb(err);
            }
            answer = JSON.parse(response.body);
            if (answer.message == null) {
              return cb();
            }
            if (answer.message !== &#039;Not Found&#039;) {
              return cb(answer.message);
            }
            console.log(&quot;Create new GitHub repository&quot;);
            debug(&quot;POST https://api.github.com/user/repos&quot;);
            return request({
              uri: &quot;https://api.github.com/user/repos&quot;,
              auth: {
                user: input.username,
                pass: input.password
              },
              headers: {
                &#039;User-Agent&#039;: input.username
              },
              method: &#039;POST&#039;,
              body: JSON.stringify({
                name: gitname,
                description: input.description,
                homepage: &quot;http://&quot; + input.username + &quot;.github.io/&quot; + gitname,
                &quot;private&quot;: false,
                has_issues: true,
                has_wiki: false,
                has_downloads: true
              })
            }, function(err, response, body) {
              if (err) {
                return cb(err);
              }
              if ((response != null ? response.statusCode : void 0) !== 201) {
                return cb(&quot;GitHub status was &quot; + response.statusCode + &quot; in try to create repository&quot;);
              }
              console.log(&quot;Connect with GitHub repository&quot;);
              debug(&quot;exec &quot; + dir + &quot;&gt; git remote add origin &quot; + options.github);
              return execFile(&quot;git&quot;, [&#039;remote&#039;, &#039;add&#039;, &#039;origin&#039;, options.github], {
                cwd: dir
              }, function(err, stdout, stderr) {
                if (stdout &amp;&amp; options.verbose) {
                  console.log(chalk.grey(stdout.trim()));
                }
                if (stderr) {
                  console.error(chalk.magenta(stderr.trim()));
                }
                return cb(err);
              });
            });
          });
        });
      });
    });
  };

  createPackage = function(dir, options, cb) {
    var file, gitname, gituser, pack, ref, ref1, ref2, ref3;
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for existing package.json&quot;));
    }
    file = path.join(dir, &#039;package.json&#039;);
    if (fs.existsSync(file)) {
      console.log(chalk.yellow(&quot;Skipped package.json creation, because already exists&quot;));
      return cb();
    }
    console.log(&quot;Create new package.json file&quot;);
    gitname = path.basename(dir);
    gituser = path.basename(path.dirname(options.github));
    pack = {
      name: options[&quot;package&quot;],
      version: &#039;0.0.0&#039;,
      description: &#039;&#039;,
      copyright: ((ref = (ref1 = PKG.author) != null ? ref1.name : void 0) != null ? ref : &#039;&#039;) + &quot; &quot; + ((new Date()).getFullYear()),
      &quot;private&quot;: (ref2 = options[&quot;private&quot;]) != null ? ref2 : false,
      keywords: [],
      homepage: options.github ? &quot;http://&quot; + gituser + &quot;.github.io/&quot; + gitname + &quot;/&quot; : &quot;&quot;,
      repository: {
        type: &#039;git&#039;,
        url: (ref3 = options.github) != null ? ref3 : options.git
      },
      bugs: options.github ? options.github + &quot;/issues&quot; : &quot;&quot;,
      author: PKG.author,
      contributors: [],
      license: PKG.license,
      main: &#039;./lib/index.js&#039;,
      scripts: {
        prepublish: &quot;node_modules/.bin/builder -c compile&quot;,
        test: &quot;node_modules/.bin/builder -c test&quot;
      },
      directories: {
        lib: &#039;./lib&#039;
      },
      dependencies: {},
      devDependencies: {
        &quot;alinex-builder&quot;: &quot;1.x&quot;,
        &quot;chai&quot;: &quot;2.x&quot;
      },
      optionalDependencies: {},
      engines: PKG.engines,
      os: []
    };
    return fs.writeFile(file, JSON.stringify(pack, null, 2), cb);
  };

  createReadme = function(dir, options, cb) {
    var doc, file, gitname, gituser;
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for README.md&quot;));
    }
    file = path.join(dir, &#039;README.md&#039;);
    if (fs.existsSync(file)) {
      return cb();
    }
    console.log(&quot;Create new README.md file&quot;);
    gitname = path.basename(dir);
    gituser = path.basename(path.dirname(options.github));
    doc = {
      badges: &#039;&#039;,
      install: &#039;&#039;
    };
    if (options.github) {
      doc.badges = &quot;\n[![Build Status] (https://travis-ci.org/&quot; + gituser + &quot;/&quot; + gitname + &quot;.svg?branch=master)](https://travis-ci.org/&quot; + gituser + &quot;/&quot; + gitname + &quot;) \n[![Coverage Status] (https://coveralls.io/repos/&quot; + gituser + &quot;/&quot; + gitname + &quot;/badge.png?branch=master)](https://coveralls.io/r/&quot; + gituser + &quot;/&quot; + gitname + &quot;?branch=master)&quot;;
    }
    if (!options[&quot;private&quot;]) {
      doc.badges += &quot;\n[![Dependency Status] (https://gemnasium.com/&quot; + gituser + &quot;/&quot; + gitname + &quot;.png)](https://gemnasium.com/&quot; + gituser + &quot;/&quot; + gitname + &quot;)&quot;;
      doc.install = &quot;\n[![NPM](https://nodei.co/npm/&quot; + options[&quot;package&quot;] + &quot;.png?downloads=true&amp;stars=true)](https://nodei.co/npm/&quot; + options[&quot;package&quot;] + &quot;/)&quot;;
    }
    return fs.writeFile(file, &quot;Package: &quot; + options[&quot;package&quot;] + &quot;\n=================================================\n&quot; + doc.badges + &quot;\n\nDescription comes here...\n\n\nInstall\n-------------------------------------------------\n&quot; + doc.install + &quot;\n\n\nLicense\n-------------------------------------------------\n\nCopyright &quot; + ((new Date()).getFullYear()) + &quot; Alexander Schilling\n\nLicensed under the Apache License, Version 2.0 (the \&quot;License\&quot;);\nyou may not use this file except in compliance with the License.\nYou may obtain a copy of the License at\n\n&gt;  &lt;http://www.apache.org/licenses/LICENSE-2.0&gt;\n\nUnless required by applicable law or agreed to in writing, software\ndistributed under the License is distributed on an \&quot;AS IS\&quot; BASIS,\nWITHOUT WARRANTIES OR CONDITIONS OF ANY KIND, either express or implied.\nSee the License for the specific language governing permissions and\nlimitations under the License.\n&quot;, cb);
  };

  createChangelog = function(dir, options, cb) {
    var file;
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for existing changelog&quot;));
    }
    file = path.join(dir, &#039;Changelog.md&#039;);
    if (fs.existsSync(file)) {
      return cb();
    }
    console.log(&quot;Create new changelog file&quot;);
    return fs.writeFile(file, &quot;Version changes\n=================================================\n\nThe following list gives a short overview about what is changed between\nindividual versions:\n\n&quot;, cb);
  };

  createTravis = function(dir, options, cb) {
    var coveralls, file, gituser;
    if (!options.github) {
      return cb();
    }
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for existing travis configuration&quot;));
    }
    file = path.join(dir, &#039;.travis.yml&#039;);
    if (fs.existsSync(file)) {
      return cb();
    }
    gituser = path.basename(path.dirname(options.github));
    console.log(&quot;Create new travis-ci configuration&quot;);
    console.log(chalk.yellow.bold(&quot;Log into https://travis-ci.org/profile/&quot; + gituser + &quot; and activate Travis CI&quot;));
    console.log(chalk.yellow.bold(&quot;Log into https://coveralls.io/repos/new?name=&quot; + gituser + &quot; and activate coveralls&quot;));
    console.log(chalk.yellow.bold(&quot;Log into https://gemnasium.com/projects/new_from_github and activate dependency checks&quot;));
    coveralls = &quot;COVERALLS_SERVICE_NAME=travis-ci COVERALLS_REPO_TOKEN=haQKkRgwLHbwX1dp8ltFXFTPO48c5EEWo node_modules/.bin/alinex-make test -c --coveralls&quot;;
    return fs.writeFile(file, &quot;language: node_js\nnode_js:\n   - \&quot;0.10\&quot;\n   - \&quot;0.11\&quot;\n   - \&quot;0.12\&quot;\n   - \&quot;io.js\&quot;\nafter_success:\n   - &quot; + coveralls, cb);
  };

  initialCommit = function(dir, options, cb) {
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check if git already used&quot;));
    }
    debug(&quot;exec &quot; + dir + &quot;&gt; git log&quot;);
    return execFile(&quot;git&quot;, [&#039;log&#039;], {
      cwd: dir
    }, function(err, stdout, stderr) {
      if (stdout.trim()) {
        return cb();
      }
      console.log(&quot;Initial commit&quot;);
      debug(&quot;exec &quot; + dir + &quot;&gt; git add *&quot;);
      return execFile(&quot;git&quot;, [&#039;add&#039;, &#039;*&#039;], {
        cwd: dir
      }, function(err, stdout, stderr) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        debug(&quot;exec &quot; + dir + &quot;&gt; git commit -m \&quot;Initial commit\&quot;&quot;);
        return execFile(&quot;git&quot;, [&#039;commit&#039;, &#039;-m&#039;, &#039;Initial commit&#039;], {
          cwd: dir
        }, function(err, stdout, stderr) {
          if (stdout &amp;&amp; options.verbose) {
            console.log(chalk.grey(stdout.trim()));
          }
          if (stderr) {
            console.error(chalk.magenta(stderr.trim()));
          }
          console.log(&quot;Push to origin&quot;);
          debug(&quot;exec &quot; + dir + &quot;&gt; git push origin master&quot;);
          return execFile(&quot;git&quot;, [&#039;push&#039;, &#039;origin&#039;, &#039;master&#039;], {
            cwd: dir
          }, function(err, stdout, stderr) {
            if (stdout &amp;&amp; options.verbose) {
              console.log(chalk.grey(stdout.trim()));
            }
            if (stderr) {
              console.error(chalk.magenta(stderr.trim()));
            }
            return cb();
          });
        });
      });
    });
  };

}).call(this);
