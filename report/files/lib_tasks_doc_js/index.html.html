  var Spawn, async, chalk, checkoutPages, cloneGit, createDoc, crypto, debug, exec, execFile, fs, openUrl, os, path, publish, pushOrigin, ref, updateDoc;

  Spawn = require(&#039;alinex-spawn&#039;);

  fs = require(&#039;alinex-fs&#039;);

  debug = require(&#039;debug&#039;)(&#039;builder&#039;);

  async = require(&#039;async&#039;);

  path = require(&#039;path&#039;);

  chalk = require(&#039;chalk&#039;);

  ref = require(&#039;child_process&#039;), exec = ref.exec, execFile = ref.execFile;

  os = require(&#039;os&#039;);

  crypto = require(&#039;crypto&#039;);

  module.exports.run = function(dir, options, cb) {
    var err, pack, url;
    url = path.join(dir, &#039;doc&#039;, &#039;index.html&#039;);
    if (options.watch &amp;&amp; options.browser) {
      setTimeout(function() {
        return openUrl(options, url);
      }, 5000);
    }
    try {
      pack = JSON.parse(fs.readFileSync(path.join(dir, &#039;package.json&#039;)));
    } catch (_error) {
      err = _error;
      return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
    }
    return createDoc(dir, options, function(err) {
      if (err) {
        return cb(err);
      }
      return async.parallel([
        function(cb) {
          return async.filter([path.join(GLOBAL.ROOT_DIR, &#039;var/local/docstyle&#039;, (pack.name.split(/-/))[0] + &#039;.css&#039;), path.join(GLOBAL.ROOT_DIR, &#039;var/src/docstyle&#039;, (pack.name.split(/-/))[0] + &#039;.css&#039;)], fs.exists, function(files) {
            if (!files.length) {
              return cb();
            }
            return fs.copy(files[0], path.join(dir, &#039;doc&#039;, &#039;doc-style.css&#039;), {
              overwrite: true
            }, cb);
          });
        }, function(cb) {
          return async.filter([path.join(GLOBAL.ROOT_DIR, &#039;var/local/docstyle&#039;, (pack.name.split(/-/))[0] + &#039;.js&#039;), path.join(GLOBAL.ROOT_DIR, &#039;var/src/docstyle&#039;, (pack.name.split(/-/))[0] + &#039;.js&#039;)], fs.exists, function(files) {
            if (!files.length) {
              return cb();
            }
            return fs.copy(files[0], path.join(dir, &#039;doc&#039;, &#039;doc-script.js&#039;), {
              overwrite: true
            }, cb);
          });
        }
      ], function(err) {
        if (err) {
          return cb(err);
        }
        if (options.publish) {
          return publish(dir, options, pack, cb);
        }
        if (options.browser) {
          return openUrl(options, url, cb);
        }
        return cb();
      });
    });
  };

  publish = function(dir, options, pack, cb) {
    var ref1, ref2, ref3;
    if (((ref1 = pack.scripts) != null ? ref1[&#039;doc-publish&#039;] : void 0) != null) {
      debug(&quot;exec &quot; + dir + &quot;&gt; &quot; + pack.scripts[&#039;doc-publish&#039;]);
      return exec(pack.scripts[&#039;doc-publish&#039;], {
        cwd: dir
      }, function(err, stdout, stderr) {
        if (options.verbose) {
          if (stdout) {
            console.log(chalk.grey(stdout.toString().trim()));
          }
        }
        if (stderr) {
          console.log(chalk.magenta(stderr.toString().trim()));
        }
        if (err) {
          return cb(err);
        }
        if (options.browser) {
          return openUrl(options, url, cb);
        }
        return cb();
      });
    } else if (~((ref2 = pack.repository) != null ? (ref3 = ref2.url) != null ? ref3.indexOf(&#039;github.com/&#039;) : void 0 : void 0)) {
      return fs.tempdir(null, &#039;alinex-make-&#039;, function(err, tmpdir) {
        if (err) {
          return cb(err);
        }
        return async.series([
          function(cb) {
            return cloneGit(dir, tmpdir, options, cb);
          }, function(cb) {
            return checkoutPages(dir, tmpdir, options, cb);
          }, function(cb) {
            return updateDoc(dir, tmpdir, options, cb);
          }, function(cb) {
            return pushOrigin(dir, tmpdir, options, cb);
          }
        ], function(err) {
          if (err) {
            return cb(err);
          }
          return fs.remove(tmpdir, function(err) {
            if (options.browser) {
              return openUrl(options, url, cb);
            }
            return cb();
          });
        });
      });
    } else {
      console.error(chalk.magenta(&quot;Could not publish, specify doc-publish script in package.json&quot;));
      if (options.browser) {
        return openUrl(options, url, cb);
      }
      return cb();
    }
  };

  openUrl = function(options, target, cb) {
    var opener;
    if (options.verbose) {
      console.log(chalk.grey(&quot;Open &quot; + target + &quot; in browser&quot;));
    }
    opener = (function() {
      switch (process.platform) {
        case &#039;darwin&#039;:
          return &#039;open&#039;;
        case &#039;win32&#039;:
          return &#039;start &quot;&quot;&#039;;
        default:
          return path.join(GLOBAL.ROOT_DIR, &#039;bin/xdg-open&#039;);
      }
    })();
    debug(&quot;exec&gt; &quot; + opener + &quot; \&quot;&quot; + (encodeURI(target)) + &quot;\&quot;&quot;);
    return exec(opener + &#039; &quot;&#039; + encodeURI(target) + &#039;&quot;&#039;, cb);
  };

  createDoc = function(dir, options, cb) {
    var docPath;
    docPath = path.join(dir, &#039;doc&#039;);
    return async.series([
      function(cb) {
        return fs.exists(docPath, function(exists) {
          if (!exists) {
            return cb();
          }
          if (options.verbose) {
            console.log(chalk.grey(&quot;Remove old documentation&quot;));
          }
          return fs.remove(docPath, cb);
        });
      }, function(cb) {
        if (options.verbose) {
          console.log(chalk.grey(&quot;Create new documentation folder&quot;));
        }
        return fs.mkdirs(docPath, cb);
      }, function(cb) {
        return async.parallel([
          function(cb) {
            if (options.verbose) {
              console.log(chalk.grey(&quot;Create index page&quot;));
            }
            return fs.writeFile(path.join(docPath, &#039;index.html&#039;), &quot;&lt;html&gt;\n&lt;head&gt;\n  &lt;meta http-equiv=\&quot;refresh\&quot; content=\&quot;0; url=README.md.html\&quot; /&gt;\n  &lt;script type=\&quot;text/javascript\&quot;&gt;\n      window.location.href = \&quot;README.md.html\&quot;\n  &lt;/script&gt;\n  &lt;title&gt;Page Redirection&lt;/title&gt;\n&lt;/head&gt;\n&lt;body&gt;\n  If you are not redirected automatically, follow the link to the\n  &lt;a href=&#039;README.md.html&#039;&gt;README&lt;/a&gt;.\n&lt;/body&gt;\n&lt;/html&gt;&quot;, cb);
          }, function(cb) {
            return fs.npmbin(&#039;replace&#039;, path.dirname(path.dirname(__dirname)), function(err, replace) {
              return async.series([
                function(cb) {
                  console.log(&quot;Create html documentation&quot;);
                  return fs.npmbin(&#039;docker&#039;, path.dirname(path.dirname(__dirname)), function(err, cmd) {
                    var args, proc;
                    if (err) {
                      return cb(err);
                    }
                    args = [&#039;-i&#039;, dir, &#039;-u&#039;, options.watch ? &#039;-w&#039; : &#039;&#039;, &#039;-x&#039;, &#039;.git,bin,doc,node_modules,test,lib,public,view,log,config,*/angular&#039;, &#039;-o&#039;, path.join(dir, &#039;doc&#039;), &#039;-c&#039;, &#039;autumn&#039;, &#039;--extras&#039;, &#039;fileSearch,goToLine&#039;];
                    proc = new Spawn({
                      cmd: cmd,
                      args: args,
                      check: function(proc) {
                        if (proc.code) {
                          return new Error(&quot;Got exit code of &quot; + proc.code);
                        }
                      }
                    });
                    return proc.run(cb);
                  });
                }, function(cb) {
                  var proc;
                  if (err) {
                    return cb(err);
                  }
                  if (options.verbose) {
                    console.log(chalk.grey(&quot;Correcting local links&quot;));
                  }
                  proc = new Spawn({
                    cmd: replace,
                    args: [&#039;(&lt;a href=&quot;(?!#|.+?://)[^?#&quot;]+[^/?#&quot;])(.*?&quot;)&#039;, &#039;$1.html$2&#039;, path.join(dir, &#039;doc&#039;), &#039;-r&#039;]
                  });
                  return proc.run(cb);
                }, function(cb) {
                  var pack, proc, ref1;
                  try {
                    pack = JSON.parse(fs.readFileSync(path.join(dir, &#039;package.json&#039;)));
                  } catch (_error) {
                    err = _error;
                    return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
                  }
                  if (!(((pack != null ? (ref1 = pack.repository) != null ? ref1.url : void 0 : void 0) != null) &amp;&amp; ~pack.repository.url.indexOf(&#039;github.com&#039;))) {
                    return cb();
                  }
                  proc = new Spawn({
                    cmd: replace,
                    args: [&#039;(&lt;div id=&quot;container&quot;)&gt;&#039;, &#039;$1 tabindex=&quot;0&quot;&gt;&lt;a id=&quot;fork&quot; href=&quot;&#039; + pack.repository.url + &#039;&quot; title=&quot;Fork me on GitHub&quot;&gt;&lt;/a&gt;&#039;, path.join(dir, &#039;doc&#039;), &#039;-r&#039;]
                  });
                  return proc.run(cb);
                }, function(cb) {
                  var pack, proc, ref1;
                  try {
                    pack = JSON.parse(fs.readFileSync(path.join(dir, &#039;package.json&#039;)));
                  } catch (_error) {
                    err = _error;
                    return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
                  }
                  if (!(((pack != null ? (ref1 = pack.repository) != null ? ref1.url : void 0 : void 0) != null) &amp;&amp; ~pack.repository.url.indexOf(&#039;github.com&#039;))) {
                    return cb();
                  }
                  proc = new Spawn({
                    cmd: replace,
                    args: [&#039;&lt;p&gt;(\\|)&#039;, &#039;&lt;p style=&quot;font-family:monospace;white-space:pre&quot;&gt;$1&#039;, path.join(dir, &#039;doc&#039;), &#039;-r&#039;]
                  });
                  return proc.run(cb);
                }
              ], cb);
            });
          }, function(cb) {
            var from, to;
            if (options.verbose) {
              console.log(chalk.grey(&quot;Copy images&quot;));
            }
            from = path.join(dir, &#039;src&#039;);
            to = path.join(docPath, &#039;src&#039;);
            return fs.copy(from, to, {
              include: &#039;*.{png,jpg,gif}&#039;
            }, cb);
          }
        ], cb);
      }
    ], cb);
  };

  cloneGit = function(dir, tmpdir, options, cb) {
    var err, file, pack;
    file = path.join(dir, &#039;package.json&#039;);
    try {
      pack = JSON.parse(fs.readFileSync(file));
    } catch (_error) {
      err = _error;
      return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
    }
    console.log(&quot;Cloning git repository&quot;);
    debug(&quot;exec&gt; git clone &quot; + pack.repository.url + &quot; &quot; + tmpdir);
    return execFile(&#039;git&#039;, [&#039;clone&#039;, pack.repository.url, tmpdir], function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      return cb(err);
    });
  };

  checkoutPages = function(dir, tmpdir, options, cb) {
    console.log(&quot;Checkout gh-pages branch&quot;);
    debug(&quot;exec &quot; + tmpdir + &quot;&gt; git checkout gh-pages&quot;);
    return execFile(&#039;git&#039;, [&#039;checkout&#039;, &#039;gh-pages&#039;], {
      cwd: tmpdir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (!err) {
        return cb();
      }
      debug(&quot;exec &quot; + tmpdir + &quot;&gt; git checkout --orphan gh-pages&quot;);
      return execFile(&#039;git&#039;, [&#039;checkout&#039;, &#039;--orphan&#039;, &#039;gh-pages&#039;], {
        cwd: tmpdir
      }, function(err, stdout, stderr) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        return cb(err);
      });
    });
  };

  updateDoc = function(dir, tmpdir, options, cb) {
    console.log(&quot;Update documentation&quot;);
    if (options.verbose) {
      console.log(chalk.grey(&quot;Remove old documentation&quot;));
    }
    debug(&quot;exec &quot; + tmpdir + &quot;&gt; git rm -rf .&quot;);
    return execFile(&#039;git&#039;, [&#039;rm&#039;, &#039;-rf&#039;, &#039;.&#039;], {
      cwd: tmpdir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(stdout.trim().grey);
      }
      if (stderr) {
        console.error(stderr.trim().magenta);
      }
      if (options.verbose) {
        console.log(chalk.grey(&quot;Copy new documentation into repository&quot;));
      }
      return fs.copy(path.join(dir, &#039;doc&#039;), tmpdir, function(err) {
        if (err) {
          return cb(err);
        }
        if (options.verbose) {
          console.log(chalk.grey(&quot;Add files to git&quot;));
        }
        debug(&quot;exec &quot; + tmpdir + &quot;&gt; git add *&quot;);
        return execFile(&#039;git&#039;, [&#039;add&#039;, &#039;*&#039;], {
          cwd: tmpdir
        }, function(err, stdout, stderr) {
          if (stdout &amp;&amp; options.verbose) {
            console.log(stdout.trim().grey);
          }
          if (stderr) {
            console.error(stderr.trim().magenta);
          }
          if (err) {
            return cb(err);
          }
          if (options.verbose) {
            console.log(chalk.grey(&quot;Commit changes&quot;));
          }
          debug(&quot;exec &quot; + tmpdir + &quot;&gt; git commit -m \&quot;Updated documentation\&quot;&quot;);
          return execFile(&#039;git&#039;, [&#039;commit&#039;, &#039;-m&#039;, &quot;Updated documentation&quot;], {
            cwd: tmpdir
          }, function(err, stdout, stderr) {
            if (stdout &amp;&amp; options.verbose) {
              console.log(chalk.grey(stdout.trim()));
            }
            if (stderr) {
              console.error(chalk.magenta(stderr.trim()));
            }
            return cb(err);
          });
        });
      });
    });
  };

  pushOrigin = function(dir, tmpdir, options, cb) {
    console.log(&quot;Push to git origin&quot;);
    debug(&quot;exec &quot; + tmpdir + &quot;&gt; git push origin gh-pages&quot;);
    return execFile(&quot;git&quot;, [&#039;push&#039;, &#039;origin&#039;, &#039;gh-pages&#039;], {
      cwd: tmpdir
    }, function(err, stdout, stderr) {
      if (stdout &amp;&amp; options.verbose) {
        console.log(chalk.grey(stdout.trim()));
      }
      if (stderr) {
        console.error(chalk.magenta(stderr.trim()));
      }
      if (err) {
        return cb(err);
      }
      return cb();
    });
  };

}).call(this);
