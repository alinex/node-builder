    <textarea id="file-source" class="col-md-12">(function() {
  var Spawn, async, chalk, commit, fs, git, path;

  Spawn = require(&#039;alinex-spawn&#039;);

  async = require(&#039;async&#039;);

  fs = require(&#039;fs&#039;);

  path = require(&#039;path&#039;);

  chalk = require(&#039;chalk&#039;);

  module.exports.run = function(dir, options, cb) {
    return async.parallel([
      function(cb) {
        return git(dir, options, cb);
      }
    ], function(err, results) {
      var i, len, r;
      if (err) {
        return cb(err);
      }
      for (i = 0, len = results.length; i &lt; len; i++) {
        r = results[i];
        if (r) {
          return cb();
        }
      }
      return cb(new Error(&quot;No supported repository to be pushed detected.&quot;));
    });
  };

  git = function(dir, options, cb) {
    if (options.verbose) {
      console.log(chalk.grey(&quot;Check for configured git&quot;));
    }
    return fs.exists(path.join(dir, &#039;.git&#039;), function(exists) {
      if (!exists) {
        return cb();
      }
      return commit(dir, options, function(err) {
        var proc;
        if (err) {
          return cb(err);
        }
        console.log(&quot;Push to origin&quot;);
        proc = new Spawn({
          cmd: &#039;git&#039;,
          args: [&#039;push&#039;, &#039;--tags&#039;, &#039;origin&#039;, &#039;master&#039;],
          cwd: dir
        });
        return proc.run(function(err, stdout, stderr) {
          if (stdout &amp;&amp; options.verbose) {
            console.log(chalk.grey(stdout.trim()));
          }
          if (stderr) {
            console.error(chalk.magenta(stderr.trim()));
          }
          return cb(err, true);
        });
      });
    });
  };

  commit = function(dir, options, cb) {
    var proc;
    if (options.message) {
      console.log(&quot;Adding changed files to git&quot;);
      proc = new Spawn({
        cmd: &#039;git&#039;,
        args: [&#039;add&#039;, &#039;-A&#039;],
        cwd: dir
      });
      return proc.run(function(err, stdout, stderr) {
        if (stdout &amp;&amp; options.verbose) {
          console.log(chalk.grey(stdout.trim()));
        }
        if (stderr) {
          console.error(chalk.magenta(stderr.trim()));
        }
        if (err) {
          return cb(err);
        }
        proc = new Spawn({
          cmd: &#039;git&#039;,
          args: [&#039;commit&#039;, &#039;-m&#039;, options.message],
          cwd: dir,
          check: function(proc) {
            if (proc.error) {
              return new Error(&quot;Error while pushing to git origin.&quot;);
            }
          }
        });
        return proc.run(function(err, stdout, stderr) {
          if (stdout &amp;&amp; options.verbose) {
            console.log(chalk.grey(stdout.trim()));
          }
          if (stderr) {
            console.error(chalk.magenta(stderr.trim()));
          }
          return cb(err);
        });
      });
    } else {
      proc = new Spawn({
        cmd: &#039;git&#039;,
        args: [&#039;status&#039;],
        cwd: dir,
        env: {
          LANG: &#039;C&#039;
        }
      });
      return proc.run(function(err, stdout, stderr) {
        if (err) {
          return cb(err);
        }
        if (~stdout.indexOf(&#039;nothing to commit&#039;)) {
          return cb();
        }
        console.log(stdout);
        return cb(new Error(&quot;Skipped push for &quot; + dir + &quot; because not all changes are committed, use &#039;--message &lt;message&gt;&#039;.&quot;));
      });
    }
  };

}).call(this);
