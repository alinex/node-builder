    <textarea id="file-source" class="col-md-12">(function() {
  var Spawn, async, chalk, exec, execFile, fs, npmInstall, npmOutdated, npmUpdate, path, ref, spawn, switchRepository;

  Spawn = require(&#039;alinex-spawn&#039;);

  async = require(&#039;async&#039;);

  fs = require(&#039;alinex-fs&#039;);

  path = require(&#039;path&#039;);

  chalk = require(&#039;chalk&#039;);

  ref = require(&#039;child_process&#039;), spawn = ref.spawn, exec = ref.exec, execFile = ref.execFile;

  module.exports.run = function(dir, options, cb) {
    return async.series([
      function(cb) {
        return switchRepository(dir, options, cb);
      }, function(cb) {
        return npmInstall(dir, options, cb);
      }, function(cb) {
        return npmUpdate(dir, options, cb);
      }, function(cb) {
        return npmOutdated(dir, options, cb);
      }, function(cb) {
        return switchRepository(dir, {
          registry: &#039;https://registry.npmjs.org/&#039;
        }, cb);
      }
    ], cb);
  };

  switchRepository = function(dir, options, cb) {
    var err, file, pack, proc, ref1, registry;
    file = path.join(dir, &#039;package.json&#039;);
    try {
      pack = JSON.parse(fs.readFileSync(file));
    } catch (_error) {
      err = _error;
      return cb(new Error(&quot;Could not load &quot; + file + &quot; as valid JSON.&quot;));
    }
    registry = options.registry ? options.registry : (ref1 = pack.publishConfig) != null ? ref1.registry : void 0;
    if (registry == null) {
      registry = &#039;https://registry.npmjs.org/&#039;;
    }
    proc = new Spawn({
      priority: 9,
      cmd: &#039;npm&#039;,
      args: [&#039;set&#039;, &#039;registry&#039;, registry],
      cwd: dir,
      input: &#039;inherit&#039;,
      check: function(proc) {
        if (proc.code) {
          return new Error(&quot;Got exit code of &quot; + proc.code);
        }
      }
    });
    return proc.run(cb);
  };

  npmInstall = function(dir, options, cb) {
    var proc;
    console.log(&quot;Install through npm&quot;);
    proc = new Spawn({
      priority: 9,
      cmd: &#039;npm&#039;,
      args: [&#039;install&#039;],
      cwd: dir,
      input: &#039;inherit&#039;,
      check: function(proc) {
        if (proc.code) {
          return new Error(&quot;Got exit code of &quot; + proc.code);
        }
      }
    });
    return proc.run(cb);
  };

  npmUpdate = function(dir, options, cb) {
    var proc;
    console.log(&quot;Update npm packages&quot;);
    proc = new Spawn({
      priority: 9,
      cmd: &#039;npm&#039;,
      args: [&#039;update&#039;],
      cwd: dir,
      input: &#039;inherit&#039;,
      check: function(proc) {
        if (proc.code) {
          return new Error(&quot;Got exit code of &quot; + proc.code);
        }
      }
    });
    return proc.run(cb);
  };

  npmOutdated = function(dir, options, cb) {
    var proc;
    console.log(&quot;List outdated packages&quot;);
    proc = new Spawn({
      priority: 9,
      cmd: &#039;npm&#039;,
      args: [&#039;outdated&#039;],
      cwd: dir,
      input: &#039;inherit&#039;,
      check: function(proc) {
        if (proc.code) {
          return new Error(&quot;Got exit code of &quot; + proc.code);
        }
      }
    });
    return proc.run(function(err, stdout) {
      var i, len, line, num, ref1;
      num = -1;
      ref1 = stdout.split(/\n/);
      for (i = 0, len = ref1.length; i &lt; len; i++) {
        line = ref1[i];
        if (!line || line.match(/\s&gt;\s/)) {
          continue;
        }
        console.log(line);
        num++;
      }
      if (num &gt; 0) {
        return cb(new Error(&quot;You may upgrade the listet modules.&quot;));
      }
      console.log(chalk.grey(&quot;Nothing to upgrade in this package found.&quot;));
      return cb();
    });
  };

}).call(this);
